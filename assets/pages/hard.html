            shopping: {

Skriv en funktion addItem(listKey, itemText) som:
Finder den rigtige liste i todoApp via listKey
Tilføjer en ny item med text fra argumentet og done: false 


                       
            
                function addItem(listKey, itemText) {
                const list = todoApp[listKey];
                if (list) {
                const newItem = {text: itemText, done: false };
                list.items.push(newItem);
                }
                console.log(todoApp);
                }

                addItem("shopping", "Køb smør");

Opgave 3

Skriv en funktion toggleItemDone(listKey, itemNr) som:
Finder den rigtige liste og item Skifter done fra true til false eller omvendt 


                       
            
                
                function toggleItemDone(listKey, itemNr) {
                const list = todoApp[listKey];
                if (list) {
                    const item = list.items[itemNr];
                    if (item) {
                    item.done = !item.done;
                    }
                }
                    console.log(todoApp);
                }

                toggleItemDone("school", 1);
